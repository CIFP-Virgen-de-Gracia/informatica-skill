SELECT DISTINCT ?langLabel ?humanLabel ?picture ?langDate WHERE {
  ?lang (wdt:P31/(wdt:P279*)) wd:Q9143.
  ?human wdt:P31 wd:Q5;
    wdt:P18 ?picture.
  { ?lang wdt:P287 ?human. }
  UNION
  { ?lang wdt:P170 ?human. }
  UNION
  { ?lang wdt:P943 ?human. }
  UNION
  { ?lang wdt:P178 ?human. }
  SERVICE wikibase:label { bd:serviceParam wikibase:language "en". }
  OPTIONAL { ?lang wdt:P571 ?langDate. }
}
LIMIT 10
